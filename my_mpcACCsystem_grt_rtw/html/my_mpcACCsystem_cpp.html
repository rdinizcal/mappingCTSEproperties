502    my_mpcACCsystem_B.DataTypeConv2_l = static_cast<int8_T>(u < 0.0 ?



503      static_cast<int32_T>(static_cast<int8_T>(-static_cast<int8_T>



504      (static_cast<uint8_T>(-u)))) : static_cast<int32_T>(static_cast<int8_T>(



505      static_cast<uint8_T>(u))));



506



507    /* RelationalOperator: '<S81>/Equal1' */



508    my_mpcACCsystem_B.Equal1_l = (my_mpcACCsystem_B.DataTypeConv1_b ==



509      my_mpcACCsystem_B.DataTypeConv2_l);



510



511    /* Logic: '<S81>/AND3' */



512    my_mpcACCsystem_B.AND3_i = (my_mpcACCsystem_B.NotEqual_o &&



513      my_mpcACCsystem_B.Equal1_l);



514    if (rtmIsMajorTimeStep((&my_mpcACCsystem_M))) {



515      /* Memory: '<S81>/Memory' */



516      my_mpcACCsystem_B.Memory_j = my_mpcACCsystem_DW.Memory_PreviousInput_d;



517    }



518



519    /* Switch: '<S81>/Switch' */



520    if (my_mpcACCsystem_B.Memory_j) {



521      /* Switch: '<S81>/Switch' incorporates:



522       *  Constant: '<S81>/Constant1'



523       */



524      my_mpcACCsystem_B.Switch_f = my_mpcACCsystem_P.Constant1_Value_k;



525    } else {



526      /* Switch: '<S81>/Switch' */



527      my_mpcACCsystem_B.Switch_f = my_mpcACCsystem_B.IntegralGain_j;



528    }



529



530    /* End of Switch: '<S81>/Switch' */



531



532    /* Saturate: '<S97>/Saturation' */



533    u = my_mpcACCsystem_B.Sum_b;



534    u1 = my_mpcACCsystem_P.amin_ego;



535    u2 = my_mpcACCsystem_P.amax_ego;



536    if (u > u2) {



537      /* Saturate: '<S97>/Saturation' */



538      my_mpcACCsystem_B.Saturation_e = u2;



539    } else if (u < u1) {



540      /* Saturate: '<S97>/Saturation' */



541      my_mpcACCsystem_B.Saturation_e = u1;



542    } else {



543      /* Saturate: '<S97>/Saturation' */



544      my_mpcACCsystem_B.Saturation_e = u;



545    }



546



547    /* End of Saturate: '<S97>/Saturation' */



548



549    /* Switch: '<S1>/Switch' */



550    if (my_mpcACCsystem_B.y > my_mpcACCsystem_P.Switch_Threshold) {



551      /* Switch: '<S1>/Switch' */



552      my_mpcACCsystem_B.a_ego = my_mpcACCsystem_B.Saturation;



553    } else {



554      /* Switch: '<S1>/Switch' */



555      my_mpcACCsystem_B.a_ego = my_mpcACCsystem_B.Saturation_e;



556    }



557



558    /* End of Switch: '<S1>/Switch' */



559



560    /* Integrator: '<S2>/Integrator' */



561    my_mpcACCsystem_B.Integrator_b = my_mpcACCsystem_X.Integrator_CSTATE_a;



562



563    /* Integrator: '<S3>/Integrator' */



564    my_mpcACCsystem_B.Integrator_d = my_mpcACCsystem_X.Integrator_CSTATE_i;



565



566    /* TransferFcn: '<S3>/Transfer Fcn' */



567    my_mpcACCsystem_B.TransferFcn_m = 0.0;



568    my_mpcACCsystem_B.TransferFcn_m += my_mpcACCsystem_P.TransferFcn_C_f *



569      my_mpcACCsystem_X.TransferFcn_CSTATE_a;



570



571    /* Sum: '<S3>/Sum1' incorporates:



572     *  Constant: '<Root>/v0 lead'



573     */



574    my_mpcACCsystem_B.Sum1_a = my_mpcACCsystem_B.TransferFcn_m +



575      my_mpcACCsystem_P.v0_lead;



576    if (rtmIsMajorTimeStep((&my_mpcACCsystem_M))) {



577      /* Sin: '<Root>/Sine Wave' */



578      if (my_mpcACCsystem_DW.systemEnable != 0) {



579        my_mpcACCsystem_DW.lastSin = std::sin(my_mpcACCsystem_P.SineWave_Freq *



580          ((((&my_mpcACCsystem_M)->Timing.clockTick1+(&my_mpcACCsystem_M)



581             ->Timing.clockTickH1* 4294967296.0)) * 0.1));



582        my_mpcACCsystem_DW.lastCos = std::cos(my_mpcACCsystem_P.SineWave_Freq *



583          ((((&my_mpcACCsystem_M)->Timing.clockTick1+(&my_mpcACCsystem_M)



584             ->Timing.clockTickH1* 4294967296.0)) * 0.1));



585        my_mpcACCsystem_DW.systemEnable = 0;



586      }



587



588      /* Sin: '<Root>/Sine Wave' */



589      my_mpcACCsystem_B.a_lead = ((my_mpcACCsystem_DW.lastSin *



590        my_mpcACCsystem_P.SineWave_PCos + my_mpcACCsystem_DW.lastCos *



591        my_mpcACCsystem_P.SineWave_PSin) * my_mpcACCsystem_P.SineWave_HCos +



592        (my_mpcACCsystem_DW.lastCos * my_mpcACCsystem_P.SineWave_PCos -



593         my_mpcACCsystem_DW.lastSin * my_mpcACCsystem_P.SineWave_PSin) *



594        my_mpcACCsystem_P.SineWave_Hsin) * my_mpcACCsystem_P.A_acc_lead +



595        my_mpcACCsystem_P.SineWave_Bias;



596    }



597  }



598



599  if (rtmIsMajorTimeStep((&my_mpcACCsystem_M))) {



600    /* Matfile logging */



601    rt_UpdateTXYLogVars((&my_mpcACCsystem_M)->rtwLogInfo, ((&my_mpcACCsystem_M



602      )->Timing.t));



603  }                                    /* end MajorTimeStep */



604



605  if (rtmIsMajorTimeStep((&my_mpcACCsystem_M))) {



606    real_T HoldCosine;



607    real_T HoldSine;



608    if (rtmIsMajorTimeStep((&my_mpcACCsystem_M))) {



609      /* Update for Memory: '<S31>/Memory' */



610      my_mpcACCsystem_DW.Memory_PreviousInput = my_mpcACCsystem_B.AND3;



611



612      /* Update for Memory: '<S81>/Memory' */



613      my_mpcACCsystem_DW.Memory_PreviousInput_d = my_mpcACCsystem_B.AND3_i;



614



615      /* Update for Sin: '<Root>/Sine Wave' */



616      HoldSine = my_mpcACCsystem_DW.lastSin;



617      HoldCosine = my_mpcACCsystem_DW.lastCos;



618      my_mpcACCsystem_DW.lastSin = HoldSine * my_mpcACCsystem_P.SineWave_HCos +



619        HoldCosine * my_mpcACCsystem_P.SineWave_Hsin;



620      my_mpcACCsystem_DW.lastCos = HoldCosine * my_mpcACCsystem_P.SineWave_HCos



621        - HoldSine * my_mpcACCsystem_P.SineWave_Hsin;



622    }



623  }                                    /* end MajorTimeStep */



624



625  if (rtmIsMajorTimeStep((&my_mpcACCsystem_M))) {



626    /* signal main to stop simulation */



627    {                                  /* Sample time: [0.0s, 0.0s] */



628      if ((rtmGetTFinal((&my_mpcACCsystem_M))!=-1) &&



629          !((rtmGetTFinal((&my_mpcACCsystem_M))-((((&my_mpcACCsystem_M)



630               ->Timing.clockTick1+(&my_mpcACCsystem_M)->Timing.clockTickH1*



631               4294967296.0)) * 0.1)) > ((((&my_mpcACCsystem_M)



632              ->Timing.clockTick1+(&my_mpcACCsystem_M)->Timing.clockTickH1*



633              4294967296.0)) * 0.1) * (DBL_EPSILON))) {



634        rtmSetErrorStatus((&my_mpcACCsystem_M), "Simulation finished");



635      }



636    }



637



638    rt_ertODEUpdateContinuousStates(&(&my_mpcACCsystem_M)->solverInfo);



639



640    /* Update absolute time for base rate */



641    /* The "clockTick0" counts the number of times the code of this task has



642     * been executed. The absolute time is the multiplication of "clockTick0"



643     * and "Timing.stepSize0". Size of "clockTick0" ensures timer will not



644     * overflow during the application lifespan selected.



645     * Timer of this task consists of two 32 bit unsigned integers.



646     * The two integers represent the low bits Timing.clockTick0 and the high bits



647     * Timing.clockTickH0. When the low bit overflows to 0, the high bits increment.



648     */



649    if (!(++(&my_mpcACCsystem_M)->Timing.clockTick0)) {



650      ++(&my_mpcACCsystem_M)->Timing.clockTickH0;



651    }



652



653    (&my_mpcACCsystem_M)->Timing.t[0] = rtsiGetSolverStopTime



654      (&(&my_mpcACCsystem_M)->solverInfo);



655



656    {



657      /* Update absolute timer for sample time: [0.1s, 0.0s] */



658      /* The "clockTick1" counts the number of times the code of this task has



659       * been executed. The resolution of this integer timer is 0.1, which is the step size



660       * of the task. Size of "clockTick1" ensures timer will not overflow during the



661       * application lifespan selected.



662       * Timer of this task consists of two 32 bit unsigned integers.



663       * The two integers represent the low bits Timing.clockTick1 and the high bits



664       * Timing.clockTickH1. When the low bit overflows to 0, the high bits increment.



665       */



666      (&my_mpcACCsystem_M)->Timing.clockTick1++;



667      if (!(&my_mpcACCsystem_M)->Timing.clockTick1) {



668        (&my_mpcACCsystem_M)->Timing.clockTickH1++;



669      }



670    }



671  }                                    /* end MajorTimeStep */



672}
